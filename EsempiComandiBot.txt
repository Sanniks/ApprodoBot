#EVENTI


@bot.event
#INIZIALIZZA IL BOT
async def on_ready():
    print(f"{bot.user.name} √® online!")

@bot.event
async def on_message(message):
#Legge tutti i messaggi
    if message.author == bot.user:
    #Ignora i propri messaggi
        return
    
    if "shit" in message.content.lower():
    #Se qualcuno scrive la parola "shit", il messaggio viene cancellato e l'utente rimproverato.
        await message.delete()
        await message.channel.send(f"{message.author.mention}, hai usato una parola bannata!")
    
    await bot.process_commands(message) #Equivalente a un while loop

@bot.event
async def on_member_join(member):
    #Comportamento all'ingresso di un membro
    await member.send(f"Benvenuto ad Approdo {member.name}!")



#COMANDI

@bot.command()
async def assign(ctx):
#Assegna un ruolo
    role = discord.utils.get(ctx.guild.roles, name=secret_role)
    if role:
        await ctx.author.add_roles(role)
        await ctx.send(f"{ctx.author.mention} is now assigned to {secret_role}")
    else:
        await ctx.send("Role doesn't exist")

@bot.command()
async def remove(ctx):
#Rimuove un ruolo
    role = discord.utils.get(ctx.guild.roles, name=secret_role)
    if role:
        await ctx.author.remove_roles(role)
        await ctx.send(f"{ctx.author.mention} is not assigned to {secret_role} anymore")
    else:
        await ctx.send("Role doesn't exist")

@bot.command()
async def dm(ctx, *, msg):
#Manda un messaggio in privato
    await ctx.author.send(f"You said {msg}")

@bot.command()
async def reply(ctx):
#Risponde al tuo messaggio
    await ctx.reply("This is a reply to your message!")

@bot.command()
@commands.has_role(secret_role) #Comando protetto, utilizzabile solo se hai il secretrole

    await ctx.send("Welcome to the club! Shhhh")

@secret.error #Come ti risponde se provi a usare un comando protetto, ma non hai il secret_role
